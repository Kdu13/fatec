using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

    {
        static void Main(string[] args)
        {
            string texto = "gege;caio;carlos,pedro";

            foreach (string nome in nomes)

                Thread.Sleep(500);
            }


            Console.ReadKey();
        }
    }
