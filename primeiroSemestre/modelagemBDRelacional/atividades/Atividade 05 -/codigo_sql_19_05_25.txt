CREATE TABLE CONSUMIDOR
(cd_consumidor INTEGER primary key,
 nm_consumidor VARCHAR(50) not null,
 nm_endereco_consumidor VARCHAR(100));

CREATE TABLE MERCADORIA
(cd_mercadoria INTEGER primary key,
nm_mercadoria VARCHAR(50) not null,
vl_unitario_mercadoria NUMBER(10,2) DEFAULT '20.50')

CREATE TABLE NOTA_FISCAL
(cd_nota_fiscal INTEGER primary key,
 dt_emissao_nota_fiscal DATE DEFAULT TO_DATE('2025-05-19', 'YYYY-MM-DD'),
 vl_total_nota_fiscal NUMBER(10,2),
 CONSUMIDOR_cd_consumidor INTEGER,
 CONSTRAINT CONSUMIDOR_cd_consumidor_fk FOREIGN KEY (CONSUMIDOR_cd_consumidor) REFERENCES CONSUMIDOR (cd_consumidor))

CREATE TABLE MERCADORIA_NOTA_FISCAL
(NOTA_FISCAL_cd_nota_fiscal INTEGER,
 MERCADORIA_cd_mercadoria INTEGER,
 qt_mercadoria INTEGER,
 vl_total_mercadoria_nota NUMBER(10,2),
 CONSTRAINT MERCADORIA_NOTA_FISCAL_PK PRIMARY KEY (NOTA_FISCAL_cd_nota_fiscal, MERCADORIA_cd_mercadoria),
 CONSTRAINT MERCADORIA_NOTA_FISCAL_NOTA_FISCAL_FK FOREIGN KEY (NOTA_FISCAL_cd_nota_fiscal) REFERENCES NOTA_FISCAL (cd_nota_fiscal),
 CONSTRAINT MERCADORIA_NOTA_FISCAL_MERCADORIA_FK FOREIGN KEY (MERCADORIA_cd_mercadoria) REFERENCES MERCADORIA (cd_mercadoria))
