-- Tabela de Jogadores
CREATE TABLE Jogador (
    id_jogador NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(100) NOT NULL,
    email VARCHAR2(100)
);

-- Tabela de Classes
CREATE TABLE Classe (
    id_classe NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(50) NOT NULL,
    descricao CLOB
);

-- Tabela de Personagens
CREATE TABLE Personagem (
    id_personagem NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(100) NOT NULL,
    nivel NUMBER DEFAULT 1,
    raca VARCHAR2(50),
    jogador_id NUMBER,
    classe_id NUMBER,
    CONSTRAINT fk_jogador FOREIGN KEY (jogador_id) REFERENCES Jogador(id_jogador),
    CONSTRAINT fk_classe FOREIGN KEY (classe_id) REFERENCES Classe(id_classe)
);

-- Tabela de Armas Lendárias (Relacionamento 1:1 com Personagem)
CREATE TABLE Arma_Lendaria (
    id_arma NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(100),
    descricao CLOB,
    dano VARCHAR2(20),
    personagem_id NUMBER UNIQUE,
    CONSTRAINT fk_personagem_arma FOREIGN KEY (personagem_id) REFERENCES Personagem(id_personagem)
);

-- Tabela de Sessões de jogo
CREATE TABLE Sessao (
    id_sessao NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    data DATE,
    descricao CLOB
);

-- Tabela de relacionamento Personagem-Sessao (N:N)
CREATE TABLE Personagem_Sessao (
    personagem_id NUMBER,
    sessao_id NUMBER,
    xp_ganho NUMBER,
    CONSTRAINT pk_personagem_sessao PRIMARY KEY (personagem_id, sessao_id),
    CONSTRAINT fk_ps_personagem FOREIGN KEY (personagem_id) REFERENCES Personagem(id_personagem),
    CONSTRAINT fk_ps_sessao FOREIGN KEY (sessao_id) REFERENCES Sessao(id_sessao)
);
