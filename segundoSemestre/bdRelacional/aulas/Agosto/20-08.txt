SELECT UPPER(r.region_name), LOWER(c.country_name)
FROM regions r, countries c
WHERE c.region_id = r.region_id
 
SELECT UPPER(r.region_name), LOWER(c.country_name)
FROM regions r JOIN countries c
ON c.region_id = r.region_id
 
SELECT UPPER(r.region_name), LOWER(c.country_name)
FROM regions r JOIN countries c
USING (region_id)
 
SELECT UPPER(r.region_name), LOWER(c.country_name)
FROM regions r NATURAL JOIN countries c
 
SELECT UPPER(r.region_name), LOWER(c.country_name)
FROM regions r, countries c
WHERE c.region_id = r.region_id AND r.region_name = 'Europe'
 
Select UPPER(r.region_name), LOWER(c.country_name)
FROM regions r JOIN countries c
ON (c.region_id = r.region_id)
where region_name = 'Europe'
 
SELECT UPPER(r.region_name),LOWER(c.country_name) 
FROM countries c
JOIN regions r USING(region_id) WHERE region_name = 'Europe'
 
SELECT r.region_name "Continente",
    c.country_name País,
    l.city AS "Cidade"
FROM regions r, countries c, locations l
WHERE c.region_id = r.region_id 
AND l.country_id = c.country_id
 
SELECT d.department_name || ' esta localizado em '
|| l.city "Localizado Depto"
FROM departments d, locations l
WHERE d.location_id = l.location_id
 
SELECT d.department_name || ' esta localizado em '
|| l.city "Localizado Depto"
FROM departments d, locations l
WHERE d.location_id(+) = l.location_id
 
SELECT d.department_name || ' esta localizado em '
|| l.city "Localizado Depto"
FROM departments d RIGHT OUTER JOIN locations l
ON d.location_id = l.location_id
 
SELECT d.department_name || ' esta localizado em '
|| l.city "Localizado Depto"
FROM departments d LEFT OUTER JOIN locations l
USING (location_id)
 
SELECT d.department_name || ' esta localizado em '
|| l.city "Localizado Depto"
FROM departments d FULL OUTER JOIN locations l
USING (location_id)
 
SELECT J.JOB_TITLE || ' tem o funcionario ' || E.FIRST_NAME "CARGOS"
FROM EMPLOYEES E RIGHT OUTER JOIN JOBS J
ON (J.JOB_ID = E.JOB_ID)
 
 
SELECT first_name || ' ' || last_name, LENGTH(first_name),
TO_CHAR(salary, '$99,999.99')
FROM EMPLOYEES
WHERE salary BETWEEN 5000 AND 15000
ORDER BY first_name
 
SELECT first_name || ' ' || last_name, LENGTH(first_name),
TO_CHAR(salary, '$99,999.99')
FROM EMPLOYEES
WHERE salary NOT BETWEEN 5000 AND 15000
ORDER BY salary DESC
