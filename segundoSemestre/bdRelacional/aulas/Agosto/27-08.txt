SELECT EMPLOYEE_ID "ID_EMPREGADO", JOB_ID "ID_CARGO" FROM EMPLOYEES
UNION ALL
SELECT TO_NUMBER(NULL), JOB_ID FROM JOBS
ORDER BY ID_EMPREGADO;
 
 
 
SELECT FIRST_NAME, JOB_TITLE "FUNCAO E DEPTO"
FROM EMPLOYEES JOIN JOBS
USING (JOB_ID)
UNION 
SELECT FIRST_NAME, DEPARTMENT_NAME
FROM EMPLOYEES JOIN DEPARTMENTS
USING (DEPARTMENT_ID)
 
SELECT EMPLOYEE_ID "ID", JOB_ID "CARGO" 
FROM EMPLOYEES 
UNION
SELECT EMPLOYEE_ID , JOB_ID 
FROM JOB_HISTORY
ORDER BY 1
 
SELECT EMPLOYEE_ID "ID", JOB_ID "CARGO", DEPARTMENT_ID "DEPARTAMENTO"
FROM EMPLOYEES 
UNION ALL 
SELECT EMPLOYEE_ID , JOB_ID, DEPARTMENT_ID
FROM JOB_HISTORY
ORDER BY 1
