EXEMPLO:

DECLARE
/*VARIAVEL DE ENTRADA*/
    V_DEPTO INTEGER := :DEPTO;

/*VARIAVEIS DE SAIDA*/
    V_NOME VARCHAR(30);
    V_CARGO JOBS.JOB_TITLE%TYPE;

/*CRIA UM NOVO CURSOR*/
CURSOR SONINHO IS
    SELECT E.FIRST_NAME, J.JOB_TITLE
    FROM EMPLOYEES E, JOBS J
    WHERE E.DEPARTMENT_ID = V_DEPTO
    AND E.JOB_ID = J.JOB_ID;
BEGIN
    OPEN SONINHO;
LOOP

/*BUSCAR OS DADOS DO CURSOR ABERTO E COLOCAR NAS VARIAVEIS DE SAIDA NA ORDEM DO SELECT*/
    FETCH SONINHO INTO V_NOME, V_CARGO;

/*CONDIÇÃO DE SAIDA QUANDO NAO TIVER MAIS DADOS*/
EXIT WHEN SONINHO%NOTFOUND;

/*EXIBIR DADOS*/
DBMS_OUTPUT.PUT_LINE(V_NOME || ' TRABALHA COMO ' || V_CARGO);

/*FINALIZA O LOOP*/
END LOOP;

/*FECHA O CURSOR*/
CLOSE SONINHO;
END;

DECLARE 
    V_GERENTE INTEGER := :GERENTE;
    V_ID INTEGER;
    V_NOME VARCHAR(100);
    V_SALARIO NUMBER(10,2);
    V_VALE NUMBER(10,2);
CURSOR C1 IS
    SELECT FIRST_NAME || ' ' || LAST_NAME, EMPLOYEE_ID, SALARY
        FROM EMPLOYEES
    WHERE MANAGER_ID = V_GERENTE;
BEGIN
OPEN C1;
LOOP
    FETCH C1 INTO V_NOME, V_ID, V_SALARIO;
    EXIT WHEN C1%NOTFOUND;
    V_VALE := V_SALARIO * 0.05;
    INSERT INTO FUNC (ID, NOME, VALE_COMPRA)
    VALUES (V_ID, V_NOME, V_VALE);
END LOOP;
COMMIT;
CLOSE C1;
END;

        SELECT EMPLOYEE_ID, FIRST_NAME, SALARY

    COMMIT;
END;

CURSORES E REGISTROS:

EXEMPLO 1:

DECLARE 
 CURSOR EMP_CURSOR IS
 SELECT EMPLOYEE_ID, LAST_NAME FROM EMPLOYEES;
 EMP_REGISTRO EMP_CURSOR%ROWTYPE;
BEGIN
 OPEN EMP_CURSOR;
 LOOP
   FETCH EMP_CURSOR INTO EMP_REGISTRO;
   EXIT WHEN EMP_CURSOR%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(EMP_REGISTRO.EMPLOYEE_ID || ' - ' || EMP_REGISTRO.LAST_NAME);
END LOOP;
CLOSE EMP_CURSOR;
END;
    
EXEMPLO 2:

DECLARE 
 CURSOR EMP_CURSOR IS
 SELECT FIRST_NAME, DEPARTMENT_ID FROM EMPLOYEES;
BEGIN
FOR EMP_REGISTRO IN EMP_CURSOR LOOP
IF EMP_REGISTRO.DEPARTMENT_ID = 80 THEN
 DBMS_OUTPUT.PUT_LINE(EMP_REGISTRO.FIRST_NAME|| ' TRABALHA NO DEPTO. VENDAS ');
END IF;
END LOOP;
END;

EXEMPLO 3:

DECLARE 
 V_DEPT INTEGER := :DEPTO;
 V_NOME EMPLOYEES.FIRST_NAME%TYPE;
 V_JOB JOBS.JOB_TITLE%TYPE;
 CURSOR C1 IS
SELECT E.FIRST_NAME, J.JOB_TITLE
FROM EMPLOYEES E, JOBS J
WHERE E.JOB_ID = J.JOB_ID AND E.DEPARTMENT_ID = V_DEPT;
BEGIN
OPEN C1;
LOOP
 FETCH C1 INTO V_NOME, V_JOB;
 EXIT WHEN C1%NOTFOUND OR C1%ROWCOUNT > 2;
 DBMS_OUTPUT.PUT_LINE(V_NOME || ' TRABALHA COMO ' || V_JOB);
END LOOP;
CLOSE C1;
END;
