/*EXIBE QUANTO GANHA O ABEL*/

SELECT LAST_NAME, SALARY FROM EMPLOYEES
WHERE LAST_NAME = 'Abel'
 
 
/*EXIBE OS FUNCIONARIOS COM SALARIO MAIOR QUE O ABEL E QUANTO GANHAM*/

SELECT LAST_NAME, SALARY FROM EMPLOYEES
WHERE SALARY > (SELECT SALARY FROM EMPLOYEES
                WHERE LAST_NAME = 'Abel')
 
 
/*ALTERA O VALOR DO SALARIO*/
UPDATE EMPLOYEES SET SALARY = (SALARY * 0.30) + SALARY
WHERE LAST_NAME = 'Abel'
 
 
/*EXIBE O FUNCIONARIO COM ID 141*/
SELECT JOB_ID FROM EMPLOYEES
WHERE EMPLOYEE_ID = 141
 
 
/*EXIBE QUEM POSSUE O MESMO CARGO DO QUE O ID 141*/

SELECT LAST_NAME, JOB_ID, EMPLOYEE_ID FROM EMPLOYEES
WHERE JOB_ID = (SELECT JOB_ID FROM EMPLOYEES
                WHERE EMPLOYEE_ID = 141)
AND EMPLOYEE_ID != 141
 
 
/*EXIBE QUANTO GANHA O ROGERS*/
SELECT LAST_NAME, SALARY FROM EMPLOYEES
WHERE LAST_NAME = 'Rogers'
 
 
/*EXIBE QUANTO GANHA O FUNCIONARIO DE ID 141*/
SELECT SALARY FROM EMPLOYEES
WHERE EMPLOYEE_ID = 141
 
 
/*EXIBE O SOBRENOME, O ID DO CARGO, SALÁRIO, O NOME DO DEPARTAMENTO DE QUEM TEM O MESMO CARGO DE EMPREGADO DO ID 141 
E QUEM TEM O SALARIO MAIOR QUE O DO EMPREGADO COM SOBRENOME Rogers*/
SELECT E.LAST_NAME, E.JOB_ID, E.SALARY, D.DEPARTMENT_NAME FROM EMPLOYEES E, DEPARTMENTS D
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID
AND JOB_ID = (SELECT JOB_ID FROM EMPLOYEES
              WHERE EMPLOYEE_ID = 141)
AND SALARY > (SELECT SALARY FROM EMPLOYEES
              WHERE LAST_NAME = 'Rogers')
ORDER BY 1
 
 
/*EXIBE O SALARIO MINIMO*/
SELECT MIN(SALARY) FROM EMPLOYEES
 
 
/*RELACIONA A TABELA EMPLOYEES E A TABELA JOBS COM USING*/
SELECT LAST_NAME, JOB_TITLE, SALARY FROM EMPLOYEES
JOIN JOBS USING (JOB_ID)
 
 
/*EXIBE O SOBRENOME, O TITULO DO CARGO (JOB_TITLE) E O SALARIO DOS EMPREGADOS
QUE RECEBAM O SALARIO MINIMO DA EMPRESA*/
SELECT LAST_NAME, JOB_TITLE, SALARY FROM EMPLOYEES
JOIN JOBS USING (JOB_ID) 
WHERE SALARY = (SELECT MIN(SALARY) FROM EMPLOYEES)
 
 
/*EXIBE TODAS AS MEDIAS DE SALARIO*/
SELECT JOB_ID, AVG(SALARY) MEDIA FROM EMPLOYEES
GROUP BY JOB_ID ORDER BY 2 DESC
 
 
/*EXIBE A MEDIA MINIMA DE SALARIO*/
SELECT MIN(AVG(SALARY)) MEDIA FROM EMPLOYEES
GROUP BY JOB_ID
 
 
/*EXIBE O CARGO E A MEDIA MINIMA DE SALARIO*/
SELECT JOB_ID, AVG(SALARY) MEDIA FROM EMPLOYEES
GROUP BY JOB_ID HAVING AVG(SALARY) = (SELECT MIN(AVG(SALARY)) MEDIA FROM EMPLOYEES
                                      GROUP BY JOB_ID)
ORDER BY 2 DESC

/*EXIBE O SALARIO MINIMO DE TODOS OS DEPARTAMENTO POR ID*/

GROUP BY DEPARTMENT_ID
 
 
/*EXIBE O NOME, SALARIO E O ID DO DEPARTAMNETO DOS EMPREGADOS QUE GANHAM SALARIO MINIMO, 
INDEPENDENTE DO DEPARTAMENTO*/
SELECT FIRST_NAME, SALARY, DEPARTMENT_ID FROM EMPLOYEES
WHERE SALARY IN (SELECT MIN(SALARY) FROM EMPLOYEES
                 GROUP BY DEPARTMENT_ID)
 
 
/*NEGA O DE CIMA*/
SELECT FIRST_NAME, SALARY, DEPARTMENT_ID FROM EMPLOYEES
WHERE SALARY NOT IN (SELECT MIN(SALARY) FROM EMPLOYEES
                 GROUP BY DEPARTMENT_ID)
 
 
/*EXIBE TODOS OS FUNCIONARIOS QUE NÃO POSSUEM SUBORDINADOS*/
SELECT E.LAST_NAME, E.JOB_ID FROM EMPLOYEES E
WHERE E.EMPLOYEE_ID NOT IN (SELECT G.MANAGER_ID FROM EMPLOYEES G
                            WHERE G.MANAGER_ID IS NOT NULL)
 
 
/*O CONTRARIO DO DE CIMA*/
SELECT E.LAST_NAME, E.JOB_ID FROM EMPLOYEES E
WHERE E.EMPLOYEE_ID IN (SELECT G.MANAGER_ID FROM EMPLOYEES G
                        WHERE G.MANAGER_ID IS NOT NULL)
 
 
/*EXIBE O SALARIO DOS PROGRAMADORES DE TI DA EMPRESA*/
SELECT SALARY FROM EMPLOYEES
WHERE JOB_ID = UPPER('it_prog')
ORDER BY SALARY DESC
 
 
/*EXIBE O ID DP EMPREGADO, SOBRENOME, ID DO CARGO E O SALARIO  DOS FUNCIONARIOS QUE GANHAM
MENOS QUE O MAIOR SALARIO DO EMPREGADO COM ID DE PROGRAMADOR DE TI E QUE NÃO TENHA ESTA FUNÇÃO NA EMPRESA*/
SELECT EMPLOYEE_ID, LAST_NAME, JOB_ID, SALARY FROM EMPLOYEES
WHERE SALARY < ANY (SELECT SALARY FROM EMPLOYEES
                     WHERE JOB_ID = 'IT_PROG')
 
 
/*A MESMA DO DE CIMA MAS EXCLUI OS PROGRAMADORES DE TI*/
SELECT EMPLOYEE_ID, LAST_NAME, JOB_ID, SALARY FROM EMPLOYEES
WHERE SALARY < ANY (SELECT SALARY FROM EMPLOYEES
                     WHERE JOB_ID = 'IT_PROG')
AND JOB_ID <> 'IT_PROG'
 
 
/*EXIBE O MENOR QUE O MENOR SALARIO*/
SELECT EMPLOYEE_ID, LAST_NAME, JOB_ID, SALARY FROM EMPLOYEES
WHERE SALARY < ALL (SELECT SALARY FROM EMPLOYEES
                     WHERE JOB_ID = 'IT_PROG')
AND JOB_ID <> 'IT_PROG'
