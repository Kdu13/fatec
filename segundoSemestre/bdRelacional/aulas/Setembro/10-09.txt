/* Subconsulta Avançada */
SELECT E.EMPLOYEE_ID, E.JOB_ID, E.LAST_NAME FROM EMPLOYEES E
WHERE 2 <= (SELECT COUNT(*) FROM JOB_HISTORY
            WHERE EMPLOYEE_ID = E.EMPLOYEE_ID)
 
 
/* Exibe o sobrenome, o salario, o id do depto e a média (salario medio do empregado dentro do seu depto.)
de quem ganha mais que a media no seu depto. */
SELECT E.LAST_NAME, E.SALARY, E.DEPARTMENT_ID, ROUND(M.MEDIA, 2) 
FROM EMPLOYEES E, (SELECT DEPARTMENT_ID, AVG(SALARY) MEDIA 
                   FROM EMPLOYEES
                   GROUP BY DEPARTMENT_ID) M
WHERE E.DEPARTMENT_ID = M.DEPARTMENT_ID
AND E.SALARY > M.MEDIA

/*Criar a tabela RESUMO contendo nome do departamento, quantidade e soma*/
CREATE TABLE RESUMO
(DEPT VARCHAR(30),
QTDE INTEGER,
SOMA NUMBER(10,2))
 
 
/*Utilizando subquery, inserir o nome do departamento, a quantidade e a soma salarial na tabela RESUMO*/
INSERT INTO RESUMO (SELECT D.DEPARTMENT_NAME, COUNT(E.EMPLOYEE_ID), SUM(E.SALARY)
                    FROM EMPLOYEES E, DEPARTMENTS D
                    WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID
                    GROUP BY D.DEPARTMENT_NAME)
 
 
/*Consultar os registros da tabela RESUMO:*/
SELECT * FROM RESUMO

FROM EMPLOYEES E
WHERE EXISTS (SELECT 'X' FROM EMPLOYEES
              WHERE MANAGER_ID = E.EMPLOYEE_ID)

FROM EMPLOYEES E

                        WHERE MANAGER_ID IS NOT NULL)

/* VIEW - AVANÇADA (NÃO É POSSÍVEL FAZER CRUD) */
CREATE VIEW VW_DEPT_EMP
AS SELECT D.DEPARTMENT_NAME DEPTO, COUNT(*) QTDE, SUM(E.SALARY) SOMA
FROM EMPLOYEES E, DEPARTMENTS D
WHERE D.DEPARTMENT_ID = E.DEPARTMENT_ID
GROUP BY D.DEPARTMENT_NAME
