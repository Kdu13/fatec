/* Criar uma sequência chamada seq_cliente que inicie em 2 com incremento 2 
e valor máximo de 2000 e sem valores na memória cache */
CREATE SEQUENCE SEQ_CLIENTE
START WITH 2
INCREMENT BY 2
MAXVALUE 2000
NOCACHE
 
/* Consultar a sequência criada */
SELECT * FROM USER_SEQUENCES

/* SEQUENCE: objeto criado para gerar números exclusivos que poderão ser
usados para chave primária. Obs. sublinhado é o default. Sintaxe */
CREATE SEQUENCE nomeSequencia
START WITH NÚMERO (PADRÃO 1)
INCREMENT BY NÚMERO (PADRÃO 1)
MINVALUE NÚMERO (PADRÃO 1)
MAXVALUE NÚMERO (PADRÃO 10^27)
ORDER|NOORDER
NOCACHE|CACHE NÚMERO(por default 20)
CYCLE|NOCYCLE
