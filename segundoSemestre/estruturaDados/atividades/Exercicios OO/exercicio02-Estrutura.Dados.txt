#include <iostream>
using namespace std;

class Triangulo{
   public:
   
   int a, b, c;
   
   bool verificar(){
       if(a + b > c && a + c > b && b + c > a){
           return true;
       }
       else{
            return false;
       }
   }
   
   string tipoTriangulo(){
       if(verificar()){
	if(a == b && b == c){
           return "Equilatero";
       }
       else if((a == b && b != c) || (b == c && c != a) || (a == c && a != b)){
           return "Isosceles";
       }
       else{
           return "Escaleno";
       }
    }
	else{
	   return "Nao e um triangulo";
	}
   }
   
   
       
};




int main()
{
    Triangulo obj;
    obj.a = 2;
    obj.b = 3;
    obj.c = 5;
    cout<<obj.tipoTriangulo();
    
    
    return 0;
}
