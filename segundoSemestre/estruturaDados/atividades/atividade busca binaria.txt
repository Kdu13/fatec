#include <iostream>
using namespace std;
void ordenacao(int lista[], int t){
    for(int i = 0; i < t; i++){
        for(int j = 0;j < t - i - 1;j++){
            if (lista[j]> lista[j+1]){
                int temp = lista[j];
                lista[j] = lista[j+1];
                lista[j+1]= temp;
            }
        }
    }
    cout << "Vetor Ordenado:\n";
    for(int j = 0;j < t;j++){
       cout << lista[j] << "\n";   
    }
}

int binaria(int vetor[],int n, int chave){
    int inicio = 0;
    int fim = n - 1;
    
    while(inicio <= fim){
        int meio = (inicio + fim) / 2;
        if(vetor[meio] == chave){
            return meio;
        }
        else if(vetor[meio] < chave){
            inicio = meio + 1;
        }
        else{
            fim = meio - 1;
        }
    }
    return -1;
    
}
