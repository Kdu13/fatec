CREATE TABLE DM_ALUNO
(
  nm_aluno VARCHAR(255) NOT NULL,
  nm_genero VARCHAR(255) NOT NULL,
  nr_faixa_etaria INT NOT NULL,
  id_curso INT NOT NULL,
  id_aluno INT NOT NULL,
  PRIMARY KEY (id_aluno)
);

CREATE TABLE DM_TEMPO
(
  id_tempo INT NOT NULL,
  nr_dia_semana INT NOT NULL,
  nr_dia_mes INT NOT NULL,
  nr_mes INT NOT NULL,
  nr_ano INT NOT NULL,
  nr_semestre INT NOT NULL,
  nr_feriado INT NOT NULL,
  PRIMARY KEY (id_tempo)
);

CREATE TABLE DM_DISCIPLINA
(
  id_disciplina INT NOT NULL,
  nm_disciplina VARCHAR(255) NOT NULL,
  cd_codigo INT NOT NULL,
  hr_carga_horaria INT NOT NULL,
  id_professor INT NOT NULL,
  PRIMARY KEY (id_disciplina)
);

CREATE TABLE DM_PROFESSOR
(
  id_professor INT NOT NULL,
  nm_professor VARCHAR(255) NOT NULL,
  nm_departamento VARCHAR(255) NOT NULL,
  nm_titulacao VARCHAR(255) NOT NULL,
  PRIMARY KEY (id_professor)
);

CREATE TABLE FT_DESEMPENHO
(
  id_desempenho INT NOT NULL,
  status VARCHAR(255) NOT NULL,
  nota_final NUMERIC(5,2) NOT NULL,
  media NUMERIC(5,2) NOT NULL,
  id_disciplina INT NOT NULL,
  id_aluno INT NOT NULL,
  id_tempo INT NOT NULL,
  id_professor INT NOT NULL,
  PRIMARY KEY (id_desempenho),
  FOREIGN KEY (id_disciplina) REFERENCES DM_DISCIPLINA(id_disciplina),
  FOREIGN KEY (id_aluno) REFERENCES DM_ALUNO(id_aluno),
  FOREIGN KEY (id_tempo) REFERENCES DM_TEMPO(id_tempo),
  FOREIGN KEY (id_professor) REFERENCES DM_PROFESSOR(id_professor)
);
